다음 명령을 실행하고 출력 결과에 대해 설명 하시오.

(echo START1; sleep 10; echo DONE1) &
// START1 출력, 10초 대기, DONE1 출력 
// 명령 그룹으로 묶어 실행 진행 및 백그라운드 진행 

(echo START2; sleep 10; echo DONE2) &
// START2 출력, 10초 대기, DONE2 출력 
// 명령 그룹으로 묶어 실행 진행 및 백그라운드 진행 

kill –KILL 첫 번째 프로세스
// 첫 번쨰 프로세스 종료 -> 작업 중단 

kill –STOP 첫 번째 프로세스
// 첫 번쨰 프로세스 중지  -> 프로세스 일시 중지 

kill –CONT 첫 번째 프로세스
// 첫 번쨰 프로세스 다시 실행 -> 중지된 프로세스 이어서 진행
